CREATE TABLE PATIENT (ID int PRIMARY KEY, EMAIL VARCHAR(30), PASSWORD VARCHAR(20), FIRST_NAME VARCHAR(20), LAST_NAME VARCHAR(20), NID INT, AGE INT, BLOOD_GROUP VARCHAR(10), GENDER VARCHAR(10), AREA_LOCATION VARCHAR(20), BANK_AC VARCHAR(20), BANK_BRANCH VARCHAR(20), BANK_NAME VARCHAR(20), BKASH VARCHAR(20))
CREATE TABLE DOCTOR (ID int PRIMARY KEY, EMAIL VARCHAR(30), PASSWORD VARCHAR(20), FIRST_NAME VARCHAR(20), LAST_NAME VARCHAR(20), NID INT, SPECIALISATION VARCHAR(20), CHAMBER_ADDRESS VARCHAR(50), VISITING_FEE INT, AREA_LOCATION VARCHAR(20), BANK_AC VARCHAR(20), BANK_BRANCH VARCHAR(20), BANK_NAME VARCHAR(20), BKASH VARCHAR(20))
CREATE TABLE DOCTOR_PHONE (DOC_ID int NOT NULL PRIMARY KEY, PHONE INT, FOREIGN KEY (DOC_ID) REFERENCES doctor (ID) ON DELETE CASCADE)
CREATE TABLE DOCTOR_TIME (DOC_ID int NOT NULL PRIMARY KEY, TIMINGS DATETIME, FOREIGN KEY (DOC_ID) REFERENCES doctor (ID) ON DELETE CASCADE)
CREATE TABLE DOCTOR_BOOK (DOC_ID int NOT NULL PRIMARY KEY, BOOKED_TIME DATETIME, FOREIGN KEY (DOC_ID) REFERENCES doctor (ID) ON DELETE CASCADE)
CREATE TABLE APPOINTMENT (ID int NOT NULL PRIMARY KEY, PRESCRIPTION VARCHAR(500), TESTS VARCHAR(100), DOC_ID INT, PATIENT_ID INT, FOREIGN KEY (DOC_ID) REFERENCES DOCTOR (ID) ON DELETE CASCADE, FOREIGN KEY (PATIENT_ID) REFERENCES PATIENT (ID) ON DELETE CASCADE)
CREATE TABLE DOCTOR_FREE (DOC_ID int NOT NULL PRIMARY KEY, FREE_TIME TIME, FOREIGN KEY (DOC_ID) REFERENCES doctor (ID) ON DELETE CASCADE)
CREATE TABLE PAYMENT_PORTAL (ID int NOT NULL PRIMARY KEY, TRANSACTION_MEDIUM VARCHAR(10), TRANSACTION_ID VARCHAR(100), APPOINTMENT_ID INT, FOREIGN KEY (APPOINTMENT_ID) REFERENCES appointment (ID) ON DELETE CASCADE)
CREATE TABLE HEALTH_DETAILS (PATIENT_ID int NOT NULL PRIMARY KEY, TESTS_DONE VARCHAR(100), TEST_RESULTS VARCHAR(100), FOREIGN KEY (PATIENT_ID) REFERENCES PATIENT (ID) ON DELETE CASCADE)
CREATE TABLE PATIENT_PHONE (PATIENT_ID int NOT NULL PRIMARY KEY, PHONE INT, FOREIGN KEY (PATIENT_ID) REFERENCES PATIENT (ID) ON DELETE CASCADE)
